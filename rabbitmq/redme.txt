D:\Program Files\RabbitMQ Server\rabbitmq_server-3.7.7\sbin\rabbitmqctl.bat stop

D:\Program Files\RabbitMQ Server\rabbitmq_server-3.7.7\sbin\rabbitmqctl.bat status
